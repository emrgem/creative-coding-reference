‚ö° Quick Examples

Simple Face

function setup() { createCanvas(400, 400); } function draw() { background(220); // Face fill(255, 220, 150); circle(200, 200, 150); // Eyes fill(0); circle(170, 180, 20); circle(230, 180, 20); // Smile noFill(); stroke(0); strokeWeight(3); arc(200, 200, 80, 80, 0, PI); }

Interactive Drawing

function setup() { createCanvas(400, 400); background(220); } function draw() { // Draw circles where mouse is fill(random(255), random(255), random(255)); noStroke(); circle(mouseX, mouseY, 50); }

Bouncing Ball

let x = 200; let y = 200; let xSpeed = 3; let ySpeed = 2; function setup() { createCanvas(400, 400); } function draw() { background(220); // Draw ball fill(255, 0, 0); circle(x, y, 50); // Move ball x += xSpeed; y += ySpeed; // Bounce off edges if (x > width || x < 0) { xSpeed *= -1; } if (y > height || y < 0) { ySpeed *= -1; } }
